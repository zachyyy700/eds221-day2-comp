Making vectors

# a char vector
dogs <- c("teddy", "khora", "waffle", "banjo")

typeof(dogs)
[1] "character"
# a numeric vector
weights <- c(50, 55, 25, 35)

# returns double & numeric
typeof(weights)
[1] "double"
class(weights)
[1] "numeric"
# an integer vector
dog_age <- c(5L, 6L, 1L, 7L)

# both returns integer
typeof(dog_age)
[1] "integer"
class(dog_age)
[1] "integer"
# combining classes?
dog_info <- c("teddy", 50, 5L)
dog_info
[1] "teddy" "50"    "5"    
# most flexible class in vector is char
typeof(dog_info)
[1] "character"
class(dog_info)
[1] "character"
is.character(dog_info)
[1] TRUE

Matrices

# specified by col this time
fish_size <- matrix(c(0.8, 1.2, 0.4, 0.9), ncol = 2, nrow = 2, byrow = FALSE)

fish_size
     [,1] [,2]
[1,]  0.8  0.4
[2,]  1.2  0.9
# returns type of the VALUES
typeof(fish_size)
[1] "double"
# returns matrix/array
class(fish_size)
[1] "matrix" "array" 
# combining diff types
dog_walk <- matrix(c("teddy", 5, "khora", 10), ncol = 2, nrow = 2, byrow = FALSE)

dog_walk
     [,1]    [,2]   
[1,] "teddy" "khora"
[2,] "5"     "10"   
# returns matrix/array
class(dog_walk)
[1] "matrix" "array" 
# returns char
typeof(dog_walk)
[1] "character"

Accessing matrices

whale_travel <- matrix(data = c(31.8, 1348, 46.9, 1587), nrow = 2, ncol = 2, byrow = TRUE )

whale_travel
     [,1] [,2]
[1,] 31.8 1348
[2,] 46.9 1587
# access 1348
whale_travel[1, 2]
[1] 1348
# access 46.9
whale_travel[2, 1]
[1] 46.9
# access ALL row 2
whale_travel[2, ]
[1]   46.9 1587.0
# access ALL col 1
whale_travel[, 1]
[1] 31.8 46.9
# giving only one element, "as if populated by column"
whale_travel[3]
[1] 1348

Naming list items

tacos <- list(topping = c("onion", "cilantro", "guacamole"), 
              filling = c("beans", "meat", "veggie"), 
              price = c(6.75, 8.25, 9.50))

# The whole thing
tacos
$topping
[1] "onion"     "cilantro"  "guacamole"

$filling
[1] "beans"  "meat"   "veggie"

$price
[1] 6.75 8.25 9.50
tacos[[2]]
[1] "beans"  "meat"   "veggie"
# OR
tacos$filling
[1] "beans"  "meat"   "veggie"
